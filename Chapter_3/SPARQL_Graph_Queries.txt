#Additional SPARQL queries to see examples of integration between ontology and data
#Important: THESE QUERIES WILL NOT WORK IN SNAP SPARQL. You need to use AllegroGrap, GraphDB, Stardog
#or some other tool that supports the full SPARQL specification
#These queries in particular are designed for use with Gruff or a similar SPARQL visualization tool.
#The file is called Graph Queries. Of course all SPARQL queries are Graph Queries but these are 
#designed to show interesting features combining SPARQL with a graph visualization tool like Gruff. 

# Find all the colleagues and where they are employed
SELECT  *
WHERE {?person1 :has_Colleague ?person2.
OPTIONAL {?person1 :has_Employer ?employer1}
OPTIONAL {?person2 :has_Employer ?employer2}}

#Find all the social relations
SELECT *
WHERE {?person1 :has_Social_Relationship ?person2}

#Find all the contains relations inferred by the reasoner
#This will be a messy graph
SELECT *
WHERE {?place1 :contains ?place2.}

#This uses the non-transitive sub-property contains. It gives a much cleaner graph 
SELECT *
WHERE {?place1 :contains_Direct ?place2.}

#This shows how the ontology definitions are also graphs. In this case we graph all the
#properties that have the :Employee class as their domain, as well as other properties of that property
#such as inverses and labels.
SELECT *
WHERE {?property rdfs:domain :Employee.
       ?property ?p ?o.}

#This shows the same query with a useful technique to filter out blank nodes and literal values such as strings
SELECT *
WHERE {?property rdfs:domain :Employee.
       ?property ?p ?o.
FILTER(isIri(?o) && !isBlank(?o))}
