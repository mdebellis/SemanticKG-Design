#Important: THESE QUERIES WILL NOT WORK IN SNAP SPARQL. You need to use AllegroGrap, GraphDB, Stardog
#or some other tool that supports the full SPARQL specification
#First, some fairly simple queries that aren't supported in Snap SPARQL. 

#Find all the predicates that connect Einstein and Godel
SELECT *
WHERE {:Albert_Einstein ?p :Kurt_Godel}

#Find all the predicates that connect Einstein to anything
SELECT *
WHERE {:Albert_Einstein ?p ?o}

# These queries show examples of using some of the SPARQL Path Operators such as +
# List all people that Einstein has a social relationship with
SELECT *
WHERE {:Albert_Einstein :has_Social_Relationship ?person}

#Lists the people that Einstein has a social relation with
#The * path operator includes people who have a social relation with Einstein's social relations
SELECT *
WHERE {:Albert_Einstein :has_Social_Relationship* ?person}

#Lists all the places that contain another place. 
SELECT *
WHERE {?place1 :contains ?place2}

#Lists all the places that directly contains another place. 
#E.g., for :North_America :contains :Canada :contains :British_Columbia :contains Vancover 
#Using the plus path operator means we will only see :North_America :contains :Canada
#And :Canada :contains :British_Columbia not that :North_America :contain :British_Columbia
SELECT *
WHERE {?place1 :contains+ ?place2}

SELECT *
WHERE {?person :has_Father`:has_Mother ?parent}
